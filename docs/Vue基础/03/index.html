组件概念#

<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>lesson 14</title>
<script src="https://unpkg.com/vue@next"></script>
</head>
<body>
<div id="root"></div>
</body>
<script>
// 组件的定义
// 组件具备复用性
// 全局组件，只要定义了，处处可以使用，即使不用，也会挂载，性能不高，但是使用起来简单，名字建议小写字母单词，中间用横线间隔
// 局部组件，定义了，要注册之后才能使用，性能比较高，使用起来有些麻烦，建议大些字母开头，驼峰命名
// 局部组件使用时，要做一个名字和组件间的映射对象，你不写映射，Vue 底层也会自动尝试帮你做映射
const Counter = {
data() {
return {
count: 1
}
},
template: `<div @click="count += 1">{{count}}</div>`
}
const HelloWorld = {
template: `<div>hello world</div>`
}
const app = Vue.createApp({
components: {
// counter: Counter,
// 'hello-world': HelloWorld,
Counter, HelloWorld,
},
template: `
<div>
<hello-world />
<counter />
</div>
`
});
// 全局组件
// app.component('counter-parent', {
// template: `<counter />`
// })
// app.component('counter', {
// data() {
// return {
// count: 1
// }
// },
// template: `<div @click="count += 1">{{count}}</div>`
// })
const vm = app.mount('#root');
</script>
</html>

</script>

<script>
// 单项数据流的概念: 子组件可以使用父组件传递过来的数据，但是绝对不能修改传递过来的数据。要修改的话复制一份存在自己的数据再修改。
// 若允许修改父组件传来的数据，会影响其他组件
const app = Vue.createApp({
data() {

<counter :count="num" />

<counter :count="num" />
</div>
`
});
app.component('counter', {
props: ['count'],
template: `<div @click="count += 1">{{count}}</div>`
});
const vm = app.mount('#root');
</script>

Non-Props#

<body>
<div id="root"></div>
</body>
<script>
// Non-prop 属性：单父组件传一个值给子组件，子组件不使用props接收，vue会默认把改属性添加到子组件最外层的标签上
// 在传样式时很常用
const app = Vue.createApp({
template: `
<div>
<counter msg="hello" msg1="hello1" />
</div>
`
});
app.component('counter', {
// 使用inheritAttrs可以不继承父组件传过来的值
// inheritAttrs: false,
// 除了模板，其他函数也可以接收Non-props属性，比如生命周期钩子
mounted() {
console.log(this.$attrs.msg);
},
// 当子组件最外层有多个标签时，默认不会接收父组件传过来的值
// 若要接收传过来的全部属性，可以显式声明 v-bind="$attrs"
// 若要接收某个属性，可以显式指明
template: `
<div :msg="$attrs.msg">Counter</div>
<div v-bind="$attrs">Counter</div>
<div :msg1="$attrs.msg1">Counter</div>
`
});
const vm = app.mount('#root');
</script>

父子组件间如何通过事件进行通信#

  • 子组件不能修改父组件传过来的值，但是可以通知父组件，让父组件来修改。子组件通过 $emit() 向外触发事件，父组件@监听这个事件即可。
  • 子组件向外触发时要用驼峰命名，父组件监听时要用-连接符。
<body>
<div id="root"></div>
</body>
<script>
const app = Vue.createApp({
data() {
return { count: 1 }
},
methods: {
handleAddOne(param) {
this.count += param;
}
}
template: `
<counter :count="count" @add-one="handleAddOne/>
`
});
app.component('counter', {
props: ['count],
// emits: ['addOne'],
// emits还可以对传入的参数进行校验
// emits: {
// addOne: (count) => {
// if(count > 1) {
// return true;
// }
// return false;
// }
// }
methods: {
handleClick() {
this.$emit('addOne', 2);
}
},
template: `
<div @click="handleClick">{{count}}</div>
`
});
const vm = app.mount('#root');
</script>

使用v-model简化#

(较难，了解即可)

<body>
<div id="root"></div>
</body>
<script>
const app = Vue.createApp({
data() {
return { count: 1 }
},
template: `
<counter v-model="count" />
`
});
app.component('counter', {
props: ['modelValue'],
methods: {
handleClick() {
this.$emit('update:modelValue', this.modelValue + 3);
}
},
template: `
<div @click="handleClick">{{modelValue}}</div>
`
});
const vm = app.mount('#root');
</script>

注意：这里modelValue是默认的，updata:modelValue的写法也是固定的，不能改为其他名字，若要改，需要在父组件传值时指定。指定名称后就可以写多个v-model了。

<script>
const app = Vue.createApp({
data() {
return { count: 1 }
},
template: `
<counter v-model:app="count" />
`
});
app.component('counter', {
props: ['app'],
methods: {
handleClick() {
this.$emit('update:app', this.app + 3);
}
},
template: `
<div @click="handleClick">{{app}}</div>
`
});
const vm = app.mount('#root');
</script>

组件间的双向绑定(选学)#

<body>
<div id="root"></div>
</body>
<script>
const app = Vue.createApp({
data() {
return {
count: 'a',
}
},
template: `
<counter v-model.uppercase="count" />
`
});
app.component('counter', {
props: {
'modelValue': String,
'modelModifiers': { // 传进来的修饰符
default: ()=> ({}) // 默认值为空对象，传了会变成eg. {uppercase: true}
}
},
methods: {
handleClick() {
let newValue = this.modelValue + 'b';
if(this.modelModifiers.uppercase) {
newValue = newValue.toUpperCase();
}
this.$emit('update:modelValue', newValue);
},
},
template: `
<div @click="handleClick">{{modelValue}}</div>
`
});
const vm = app.mount('#root');
</script>

作用域插槽#

  • v-slot:footer 简写成 #footer

  • Tab1
  • Tab2

动态组件和异步组件#

动态组件#

<script>
// 动态组件: 根据数据的变化，结合 compoent 这个标签，来随时动态切换组件的现实
const app = Vue.createApp({
data() {
return { currentItem: 'input-item'}
},
methods: {
handleClick() {
if(this.currentItem === 'input-item') {
this.currentItem = 'common-item';
} else {
this.currentItem = 'input-item';
}
}
},
// template: `
// <input-item v-show="currentItem === 'input-item'" />
// <common-item v-show="currentItem === 'common-item'" />
// <button @click="handleClick">切换</button>
// `
// 动态组件，加上keep-alive避免切换回输入框时里面内容被清空
template: `
<keep-alive>
<component :is="currentItem" />
</keep-alive>
<button @click="handleClick">切换</button>
`
});
app.component('input-item', {
template: `<input />`
});
app.component('common-item', {
template: `<div>hello world</div>`
})
const vm = app.mount('#root');
</script>

异步组件#

<script>
// 异步组件: 是异步执行某些组件的逻辑，这叫做异步组件
const app = Vue.createApp({
template: `
<div>
<common-item />
<async-common-item />
</div>
`
});
app.component('common-item', {
template: `<div>hello world</div>`
});
// NOTE 补充promise知识
app.component('async-common-item', Vue.defineAsyncComponent(() => {
return new Promise((resolve, reject) => {
setTimeout(() => {
resolve({
template: `<div>this is an async component</div>`
})
}, 4000)
})
}))
const vm = app.mount('#root');
</script>

  • v-once：只渲染一次，后续数据发生变化也不再渲染页面

</body>
<script>
// ref 实际上是获取Dom节点 / 组件的引用的一个语法，不建议过多地操作DOM
const app = Vue.createApp({
data() {
return { count: 1 }
},
// mounted是在页面元素加载完后执行，这时才有DOM
mounted() {
this.$refs.count.innerHTML = 'Hello';
},

<div ref="count">
{{count}}
</div>
</div>
`
});
const vm = app.mount('#root');
</script>

provide / inject#

若父组件有一个数据，孙子组件要用，则需要层层传递，如Tab1。使用 provide 后，子组件不需要帮着传，孙子组件要使用的话只需要使用 inject 就可以了。

  • Tab1
  • Tab2
<body>
<div id="root"></div>
</body>
<script>
// provide / inject
const app = Vue.createApp({
data() {
return { count: 1 }
},
template: `
<div>
<child :count="count" />
</div>
`
});
app.component('child', {
template: `<child-child :count="count"/>`
});
app.component('child-child', {
inject: ['count'],
template: `<div>{{count}}</div>`
});
const vm = app.mount('#root');
</script>
<body>
<div id="root"></div>
</body>
<script>
// provide / inject
const app = Vue.createApp({
data() {
return { count: 1 }
},
// provide: {
// count: 1,
// },
// 若要传data里的数据，需要写成函数的形式
// 这里传值是一次性的，下面的button被点击时count会+1，但是页面的数字不会发生变化，还是1
provide() {
return {
count: this.count,
}
},
template: `
<div>
<child :count="count" />
<button @click="count += 1">Add</button>
</div>
`
});
app.component('child', {
template: `<child-child />`
});
app.component('child-child', {
inject: ['count'],
template: `<div>{{count}}</div>`
});
const vm = app.mount('#root');
</script>
